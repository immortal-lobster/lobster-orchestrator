### A10-5001: AI Agent 工作流编排

**问题**: 手动编排多个 AI Agent 效率低，容易出错

**解决方案**: 使用状态机管理工作流

```python
# agent_workflow.py - AI Agent 工作流编排

from enum import Enum
from typing import Dict, Callable
import asyncio

class WorkflowState(Enum):
    PENDING = "pending"
    RUNNING = "running"
    WAITING = "waiting"  # 等待外部事件
    COMPLETED = "completed"
    FAILED = "failed"

class AgentWorkflow:
    def __init__(self):
        self.state = WorkflowState.PENDING
        self.context: Dict = {}
        self.handlers: Dict[str, Callable] = {}
    
    def register(self, name: str, handler: Callable):
        """注册 Agent 处理器"""
        self.handlers[name] = handler
    
    async def execute(self, workflow: list):
        """执行工作流"""
        self.state = WorkflowState.RUNNING
        
        for step in workflow:
            agent_name = step["agent"]
            input_data = step.get("input", {})
            timeout = step.get("timeout", 300)
            
            try:
                # 执行 Agent
                handler = self.handlers[agent_name]
                result = await asyncio.wait_for(
                    handler(input_data, self.context),
                    timeout=timeout
                )
                
                # 更新上下文
                self.context.update(result)
                
                # 检查是否需要等待
                if step.get("wait_for"):
                    self.state = WorkflowState.WAITING
                    await self.wait_for_event(step["wait_for"])
                    self.state = WorkflowState.RUNNING
                
            except asyncio.TimeoutError:
                self.state = WorkflowState.FAILED
                raise Exception(f"Agent {agent_name} 超时 ({timeout}s)")
            except Exception as e:
                self.state = WorkflowState.FAILED
                raise
        
        self.state = WorkflowState.COMPLETED
        return self.context

# 使用示例
async def main():
    workflow = AgentWorkflow()
    
    # 注册 Agent
    workflow.register("research", research_agent)
    workflow.register("write", writing_agent)
    workflow.register("review", review_agent)
    
    # 定义工作流
    steps = [
        {"agent": "research", "input": {"topic": "AI automation"}, "timeout": 600},
        {"agent": "write", "input": {"outline": "{{research.outline}}"}, "timeout": 300},
        {"agent": "review", "input": {"draft": "{{write.draft}}"}, "timeout": 180},
    ]
    
    result = await workflow.execute(steps)
    print(f"完成！生成 {len(result['draft'])} 字")
```

### A10-5003: 智能异常处理

**问题**: 自动化脚本遇到异常就崩溃，需要人工介入

**解决方案**: 自愈式异常处理

```python
# self_healing_automation.py

import time
import logging
from typing import Optional, Callable
from functools import wraps

class SelfHealingAutomation:
    def __init__(self, max_retries=3, base_delay=1):
        self.max_retries = max_retries
        self.base_delay = base_delay
        self.healing_strategies = {}
    
    def register_healing(self, exception_type: type, strategy: Callable):
        """注册自愈策略"""
        self.healing_strategies[exception_type] = strategy
    
    def auto_heal(self, func: Callable):
        """装饰器：自动异常修复"""
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            
            for attempt in range(self.max_retries + 1):
                try:
                    return func(*args, **kwargs)
                
                except Exception as e:
                    last_exception = e
                    logging.warning(f"尝试 {attempt + 1}/{self.max_retries + 1} 失败：{e}")
                    
                    # 尝试自愈
                    if attempt < self.max_retries:
                        healed = self.try_heal(e)
                        if healed:
                            logging.info("自愈成功，重试...")
                            time.sleep(self.base_delay * (2 ** attempt))  # 指数退避
                            continue
                    
                    # 无法自愈，抛出异常
                    logging.error(f"无法自愈，最终失败：{e}")
                    raise
            
            raise last_exception
        
        return wrapper
    
    def try_heal(self, exception: Exception) -> bool:
        """尝试自愈"""
        for exc_type, strategy in self.healing_strategies.items():
            if isinstance(exception, exc_type):
                try:
                    strategy(exception)
                    return True
                except Exception as e:
                    logging.error(f"自愈策略失败：{e}")
        return False

# 自愈策略示例
def refresh_browser_session(exception):
    """浏览器会话过期时刷新"""
    logging.info("刷新浏览器会话...")
    # 重新初始化浏览器
    global browser
    browser = playwright.chromium.launch()

def clear_disk_space(exception):
    """磁盘空间不足时清理"""
    import subprocess
    logging.info("清理磁盘空间...")
    subprocess.run(["rm", "-rf", "/tmp/cache/*"])

def rotate_api_key(exception):
    """API 密钥失效时轮换"""
    logging.info("轮换 API 密钥...")
    # 从密钥管理系统获取新密钥
    new_key = get_new_api_key()
    os.environ["API_KEY"] = new_key

# 使用示例
automation = SelfHealingAutomation(max_retries=3, base_delay=2)

# 注册自愈策略
automation.register_healing(SessionExpired, refresh_browser_session)
automation.register_healing(DiskFull, clear_disk_space)
automation.register_healing(APIKeyExpired, rotate_api_key)

@automation.auto_heal
def scrape_website(url: str):
    """爬取网站（带自愈能力）"""
    page = browser.new_page()
    page.goto(url)
    return page.content()

# 运行
try:
    content = scrape_website("https://example.com")
except Exception as e:
    logging.error(f"最终失败：{e}")
    # 发送告警，人工介入
    send_alert(f"自动化任务失败：{e}")
```

### A10-5101: 智能告警降噪

**问题**: 监控系统产生大量告警，运维团队告警疲劳

**解决方案**: 基于 ML 的告警聚类和优先级排序

```python
# alert_deduplication.py

import numpy as np
from sklearn.cluster import DBSCAN
from datetime import datetime, timedelta

class AlertDeduplicator:
    def __init__(self, time_window_minutes=5):
        self.time_window = timedelta(minutes=time_window_minutes)
        self.alert_history = []
    
    def add_alert(self, alert: dict):
        """添加告警到历史"""
        alert['timestamp'] = datetime.now()
        self.alert_history.append(alert)
        
        # 清理旧告警（保留 1 小时）
        cutoff = datetime.now() - timedelta(hours=1)
        self.alert_history = [
            a for a in self.alert_history 
            if a['timestamp'] > cutoff
        ]
    
    def cluster_alerts(self) -> list:
        """聚类相似告警"""
        if len(self.alert_history) < 2:
            return self.alert_history
        
        # 特征工程：将告警转换为数值向量
        vectors = []
        for alert in self.alert_history:
            vector = self.alert_to_vector(alert)
            vectors.append(vector)
        
        # DBSCAN 聚类
        clustering = DBSCAN(eps=0.5, min_samples=2).fit(vectors)
        
        # 分组告警
        clusters = {}
        for i, label in enumerate(clustering.labels_):
            if label == -1:  # 噪声点，单独处理
                clusters[f'noise_{i}'] = [self.alert_history[i]]
            else:
                if label not in clusters:
                    clusters[label] = []
                clusters[label].append(self.alert_history[i])
        
        # 生成聚合告警
        aggregated = []
        for cluster_id, alerts in clusters.items():
            if len(alerts) > 1:
                # 聚合多个相似告警
                aggregated.append({
                    'type': 'aggregated',
                    'count': len(alerts),
                    'first_seen': min(a['timestamp'] for a in alerts),
                    'last_seen': max(a['timestamp'] for a in alerts),
                    'sample_alert': alerts[0],  # 示例告警
                    'all_alerts': alerts
                })
            else:
                aggregated.append(alerts[0])
        
        return aggregated
    
    def alert_to_vector(self, alert: dict) -> np.ndarray:
        """将告警转换为数值向量（用于聚类）"""
        # 特征：服务名、错误类型、时间戳（归一化）
        service_hash = hash(alert.get('service', '')) % 1000 / 1000
        error_hash = hash(alert.get('error_type', '')) % 1000 / 1000
        time_norm = alert['timestamp'].timestamp() % 3600 / 3600  # 小时内的相对时间
        
        return np.array([service_hash, error_hash, time_norm])
    
    def prioritize_alerts(self, alerts: list) -> list:
        """优先级排序"""
        priority_scores = []
        
        for alert in alerts:
            score = 0
            
            # 因素 1: 告警数量（聚合告警优先级高）
            if alert.get('type') == 'aggregated':
                score += alert['count'] * 10
            
            # 因素 2: 服务关键性
            critical_services = ['payment', 'auth', 'database']
            if alert.get('service') in critical_services:
                score += 50
            
            # 因素 3: 错误类型
            critical_errors = ['OutOfMemory', 'DiskFull', 'ConnectionLost']
            if alert.get('error_type') in critical_errors:
                score += 30
            
            # 因素 4: 持续时间
            if alert.get('type') == 'aggregated':
                duration = (alert['last_seen'] - alert['first_seen']).total_seconds()
                score += min(duration / 60, 30)  # 最多 30 分
            
            priority_scores.append((score, alert))
        
        # 按分数降序排序
        priority_scores.sort(reverse=True, key=lambda x: x[0])
        return [alert for score, alert in priority_scores]

# 使用示例
```

## 🎯 A10-5151~5200: 自动化最佳实践

### A10-5151: 自动化 ROI 计算框架

**公式**:
```
ROI = (年节省成本 - 实施成本) / 实施成本 × 100%

年节省成本 = (人工时间节省 × 人力成本) + 错误减少损失 + 效率提升收益
实施成本 = 开发成本 + 工具许可 + 维护成本
```

**计算模板**:

```python
# roi_calculator.py

class AutomationROI:
    def __init__(self):
        self.manual_time_per_task = 0  # 分钟
        self.automated_time_per_task = 0  # 分钟
        self.tasks_per_month = 0
        self.hourly_labor_cost = 0  # $/小时
        self.error_rate_manual = 0  # %
        self.error_rate_automated = 0  # %
        self.cost_per_error = 0  # $
        self.development_cost = 0  # $
        self.tool_license_monthly = 0  # $/月
        self.maintenance_monthly = 0  # $/月
    
    def calculate_annual_savings(self) -> float:
        """计算年节省"""
        
        # 1. 时间节省
        time_saved_per_task = (self.manual_time_per_task - self.automated_time_per_task) / 60  # 小时
        time_savings_monthly = time_saved_per_task * self.tasks_per_month
        time_savings_annual = time_savings_monthly * 12
        labor_savings_annual = time_savings_annual * self.hourly_labor_cost
        
        # 2. 错误减少
        errors_manual_monthly = self.tasks_per_month * self.error_rate_manual / 100
        errors_automated_monthly = self.tasks_per_month * self.error_rate_automated / 100
        errors_reduced_monthly = errors_manual_monthly - errors_automated_monthly
        error_savings_annual = errors_reduced_monthly * 12 * self.cost_per_error
        
        # 3. 总节省
        return labor_savings_annual + error_savings_annual
    
    def calculate_implementation_cost(self) -> float:
        """计算实施成本（3 年）"""
        return (
            self.development_cost +
            (self.tool_license_monthly * 36) +
            (self.maintenance_monthly * 36)
        )
    
    def calculate_roi(self) -> float:
        """计算 ROI"""
        annual_savings = self.calculate_annual_savings()
        implementation_cost = self.calculate_implementation_cost()
        
        # 3 年总节省
        total_savings = annual_savings * 3
        
        return (total_savings - implementation_cost) / implementation_cost * 100
    
    def calculate_payback_period(self) -> float:
        """计算回收期（月）"""
        monthly_savings = self.calculate_annual_savings() / 12
        monthly_cost = self.tool_license_monthly + self.maintenance_monthly
        net_monthly_savings = monthly_savings - monthly_cost
        
        if net_monthly_savings <= 0:
            return float('inf')  # 永远无法回收
        
        return self.development_cost / net_monthly_savings
    
    def generate_report(self):
        """生成 ROI 报告"""
        print("=" * 60)
        print("自动化 ROI 分析报告")
        print("=" * 60)
        print(f"\n输入参数:")
        print(f"  手动时间：{self.manual_time_per_task} 分钟/任务")
        print(f"  自动时间：{self.automated_time_per_task} 分钟/任务")
        print(f"  任务量：{self.tasks_per_month} 任务/月")
        print(f"  人力成本：${self.hourly_labor_cost}/小时")
        print(f"  错误率：{self.error_rate_manual}% → {self.error_rate_automated}%")
        print(f"  错误成本：${self.cost_per_error}/次")
        print(f"\n实施成本:")
        print(f"  开发：${self.development_cost:,}")
        print(f"  工具：${self.tool_license_monthly:,}/月")
        print(f"  维护：${self.maintenance_monthly:,}/月")
        print(f"\n输出结果:")
        print(f"  年节省：${self.calculate_annual_savings():,.0f}")
        print(f"  3 年总节省：${self.calculate_annual_savings() * 3:,.0f}")
        print(f"  实施成本 (3 年): ${self.calculate_implementation_cost():,.0f}")
        print(f"  ROI: {self.calculate_roi():.0f}%")
        print(f"  回收期：{self.calculate_payback_period():.1f} 个月")
        print("=" * 60)
```
